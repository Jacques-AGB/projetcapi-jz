43
44 // Create and save assignment
45 if (request.getId() == null){

47 Assignment savedAssignment = assignmentRepository.save(newAssignment);
48 return assignmentMapper.toResponse(savedAssignment);
49 }else {
50 Assignment existingAssignment = assignmentRepository.findById(request.getId())
51 .orElseThrow(() -> new IllegalArgumentException("Assignment not found with id: " + request.getId()));

53 Game game = gameRepository.findById(request.getGameId())
54 .orElseThrow(() -> new IllegalArgumentException("Game not found with id: " + request.getGameId()));
55
56 // Update fields
57 existingAssignment.setLibelle(request.getLibelle());
58 existingAssignment.setDescription(request.getDescription());
59 existingAssignment.setDifficulty(request.getDifficulty());
60 existingAssignment.setGame(game);

62 // Save updated assignment
