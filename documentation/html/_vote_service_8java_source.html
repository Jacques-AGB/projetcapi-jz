34 @Override
35 public String submitVote(VoteRequest request) {
36 Optional<Assignment> assignment = assignmentRepository.findById(request.getAssignmentId());
37 if (assignment.isEmpty()) {
38 throw new IllegalArgumentException("An assignment doesn't exists.");
39 }
40 Optional<Player> playerExists = playerRepository.findById(request.getPlayerId());
41 if (playerExists.isEmpty()) {
42 throw new IllegalArgumentException("A player doesn't exists.");
43 }
44 Vote vote = new Vote();
45 vote.setAssignment(assignment.get());
46 vote.setPlayer(playerExists.get());
47 vote.setValue(request.getValue());
48
49 voteRepository.save(vote);
50 return "Player " + request.getPlayerId() + " has voted";
51 }
